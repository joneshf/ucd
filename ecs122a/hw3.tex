        \begin{algorithmic}[1]
          \Function{maxSubarray}{$arr$}
            \Let{n}{\textrm{ length of }arr}
            \Let{max}{-\infty}
            \Let{start}{0}
            \Let{end}{0}
            \For{$i \gets 0 \textrm{ to } n$}
              \Let{total}{0}
              \For{$j \gets i \textrm{ to } n$}
                \Let{total}{total + arr[j]}
                \If{$total \ge max$}
                  \Let{max}{total}
                  \Let{start}{i}
                  \Let{end}{j}
                \EndIf
              \EndFor
            \EndFor
            \State \textbf{return} $arr[start..end]$
          \EndFunction
        \end{algorithmic}
      \end{algorithm}

      N.B. If the result is expected to be a single value,
      the function should return $max$.
