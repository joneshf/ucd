      \begin{grammar}
        <room> ::= [<chairs>] <table>
                \alt <table> (couch | <stuff>)

                \alt teak
                \alt dinette
                \alt <table> <chairs>

        <chairs> ::= <chair> \{ <chair> \}

                \alt plant
      \end{grammar}

    \item
