          &= t_a t_{a''} (\rho_\theta \rho_{-\theta}) \\
          &= t_a t_{a''} (\rho_{\theta - \theta}) \\
          &= t_a t_{a''} \\
          &= t_{a + a''} \\
        \end{align*}
